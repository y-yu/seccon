	{\setlength{\fboxsep}{.4pt}%

		\colorbox{black}{\textcolor{white}{#1}}}%
		\hbox{%
			\hbox{%
				\vbox{%
					\hrule height \fboxrule\vskip-\fboxrule%
					\hbox{%
						\vrule width \fboxrule\hskip-\fboxrule%
						\box0%
						\hskip-\fboxrule%
					\vrule width \fboxrule}
					\vskip-\fboxrule}}}}}
